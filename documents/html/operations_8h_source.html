385 bool process(Window& window, Input_Status& status, b2Body* body_a, b2Body* body_b, float window_height, float scale)
386 {
387 Event event;
388
389 while (window.pollEvent(event))
390 {
391 switch (event.type)
392 {
393 case Event::KeyPressed:
394 {
395 switch (event.key.code)
396 {
397 case Keyboard::Left:
398 {
399 if (body_a->GetAngularVelocity() < +10.f) body_a->ApplyTorque(+15, true);
400 if (body_b->GetAngularVelocity() < +10.f) body_b->ApplyTorque(+15, true);
401 break;
402 }
403 case Keyboard::Right:
404 {
405 if (body_a->GetAngularVelocity() > -10.f) body_a->ApplyTorque(-10, true);
406 if (body_b->GetAngularVelocity() > -10.f) body_b->ApplyTorque(-10, true);
407 break;
408 }
409 case Keyboard::Space:
410 {
411 //Rotar la cesta
412 //...
413 break;
414 }
415 }
416
417 break;
418 }
419
420 case Event::MouseButtonPressed:
421 {
422 if (event.mouseButton.button == Mouse::Button::Left)
423 {
424 status.click_start_x = status.current_mouse_x = float(event.mouseButton.x);
425 status.click_start_y = status.current_mouse_y = float(event.mouseButton.y);
426 status.mouse_was_clicked = true;
427 }
428
429 break;
430 }
431
432 case Event::Closed:
433 {
434 return true;
435 }
436 }
437 }
438
439 return false;
440 }
