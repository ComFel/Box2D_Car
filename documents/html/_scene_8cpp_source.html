50 void Scene::render(b2World& physics_world, RenderWindow& window, float scale)
51 {
52 // Se cachea el alto de la ventana en una variable local:
53 float window_height = (float)window.getSize().y;
54
55 // Se recorre toda la lista de bodies del world:
56 for (b2Body* body = physics_world.GetBodyList(); body != nullptr; body = body->GetNext())
57 {
58 // Se obtiene el transform del body:
59 const b2Transform& body_transform = body->GetTransform();
60
61 // Se recorre la lista de fixtures del body:
62 for (b2Fixture* fixture = body->GetFixtureList(); fixture != nullptr; fixture = fixture->GetNext())
63 {
64 // Se obtiene el tipo de forma de la fixture:
65 b2Shape::Type shape_type = fixture->GetShape()->GetType();
66
67 if (shape_type == b2Shape::e_circle)
68 {
69 // Se crea un CircleShape a partir de los atributos de la forma de la fixture y del body:
70 // En SFML el centro de un círculo no está en su position. Su position es la esquina superior izquierda
71 // del cuadrado en el que está inscrito. Por eso a position se le resta el radio tanto en X como en Y.
72
73 b2CircleShape* circle = dynamic_cast<b2CircleShape*>(fixture->GetShape());
74
75 float radius = circle->m_radius * scale;
76 b2Vec2 center = circle->m_p;
77
78 CircleShape shape;
79
80 shape.setPosition(box2d_position_to_sfml_position(b2Mul(body_transform, center), window_height, scale) - Vector2f(radius, radius));
81 shape.setFillColor(Color::Red);
82 shape.setRadius(radius);
83
84 window.draw(shape);
85
86 }
87
88 if (shape_type == b2Shape::e_edge)
89 {
90 // Se toman los dos vértices del segmento y se ajusta su posición usando el transform del body.
91 // Los vértices resultantes se convierten y se ponen en un array para finalmente dibujar el segmento
92 // que los une usando la sobrecarga del método draw() que permite dibujar primitivas de OpenGL a
93 // partir de datos de vértices.
94
95 b2EdgeShape* edge = dynamic_cast<b2EdgeShape*>(fixture->GetShape());
96
97 b2Vec2 start = b2Mul(body_transform, edge->m_vertex1);
98 b2Vec2 end = b2Mul(body_transform, edge->m_vertex2);
99
100 Vertex line[] =
101 {
102 Vertex(box2d_position_to_sfml_position(start, window_height, scale), Color::Green),
103 Vertex(box2d_position_to_sfml_position(end, window_height, scale), Color::Green),
104 };
105
106 window.draw(line, 2, Lines);
107 }
108 else if (shape_type == b2Shape::e_polygon)
109 {
110 // Se toma la forma poligonal de Box2D (siempre es convexa) y se crea a partir de sus vértices un
111 // ConvexShape de SFML. Cada vértice de Box2D hay que transformarlo usando el transform del body.
112
113 b2PolygonShape* box2d_polygon = dynamic_cast<b2PolygonShape*>(fixture->GetShape());
114 ConvexShape sfml_polygon;
115
116 int number_of_vertices = box2d_polygon->m_count;
117
118 sfml_polygon.setPointCount(number_of_vertices);
119 sfml_polygon.setFillColor(Color::Green);
120
121 for (int index = 0; index < number_of_vertices; index++)
122 {
123 sfml_polygon.setPoint
124 (
125 index,
126 box2d_position_to_sfml_position(b2Mul(body_transform, box2d_polygon->m_vertices[index]), window_height, scale)
127 );
128 }
129
130 window.draw(sfml_polygon);
131 }
132 }
133 }

136 void Scene::update(b2World& physics_world, RenderWindow& window, float scale, float delta_time)
137 {
138 this->render(physics_world, window, scale);
139
140 if (doItOnce == false)
141 {
142 // Si la rueda izquierda pasa la posición de la torreta con las balls,se rota la torreta para que caigan las bolas y sale el ascensor.
143 if (entityMap["leftWheelCar"]->getBody()->GetPosition().x > entityMap["ballsColumn"]->getBody()->GetPosition().x)
144 {
145 entityMap["containerJoint"]->getBody()->SetAngularVelocity(-0.2f);
146 elevetarSeen = true;
147 doItOnce = true;
148 }
149 }
150
151 // La torreta para de girar cuando la altura de se lado derecho es menor o igual a 2.5
152 if (entityMap["rotatedBox_r"]->getBody()->GetPosition().y <= 2.5f)
153 {
154 entityMap["containerJoint"]->getBody()->SetAngularVelocity(0.f);
155 }
156
157 // El ascensor sale y se posiciona a la altura del suelo
158 if (elevetarSeen == true)
159 {
160 if (entityMap["elevator"]->getBody()->GetPosition().y < 0.10f)
161 {
162 entityMap["elevator"]->getBody()->SetLinearVelocity({ 0, +0.5f });
163 }
164 else
165 {
166 elevetarSeen = false;
167 elevetarActive = true;
168 entityMap["elevator"]->getBody()->SetLinearVelocity({ 0, +0.0f });
169 }
170 }
171
172 // Si se detecta que la reuda derecha del coche subió a la plataforma, esta sube hasta el segundo piso.
173 if (elevetarActive == true)
174 {
175 if ((entityMap["rightWheelCar"]->getBody()->GetPosition().x > 6.00f) && (entityMap["rightWheelCar"]->getBody()->GetPosition().x < 9.00f))
176 {
177 if (entityMap["elevator"]->getBody()->GetPosition().y < 4)
178 {
179 entityMap["elevator"]->getBody()->SetLinearVelocity({ 0, +0.5f });
180 }
181 else
182 {
183 entityMap["elevator"]->getBody()->SetLinearVelocity({ 0, +0.0f });
184 elevetarActive = false;
185 }
186 }
187 }
188
189 // Si el coche se cae por el vacío se resetea su posición
190 if (entityMap["carBaseShape"]->getBody()->GetPosition().y < 0.f)
191 {
192 resetCarPosition();
193 }
194 }
